---
title: Accordion
description: A vertically stacked set of interactive headings that each reveal a section of content.
---

## Installation

```bash
npm install @kuzenbo/core
```

## Usage

```tsx
import { Accordion } from "@kuzenbo/core";

function Example() {
  return (
    <Accordion defaultValue={["item-1"]}>
      <Accordion.Item value="item-1">
        <Accordion.Header>
          <Accordion.Trigger>Is it accessible?</Accordion.Trigger>
        </Accordion.Header>
        <Accordion.Content>
          Yes. It adheres to the WAI-ARIA design pattern.
        </Accordion.Content>
      </Accordion.Item>
      <Accordion.Item value="item-2">
        <Accordion.Header>
          <Accordion.Trigger>Is it styled?</Accordion.Trigger>
        </Accordion.Header>
        <Accordion.Content>
          Yes. It comes with default styles that you can customize.
        </Accordion.Content>
      </Accordion.Item>
    </Accordion>
  );
}
```

## Props

### Accordion

<ParamField path="variant" type="'default' | 'bordered' | 'ghost'" default="'default'">
  The visual style variant of the accordion.
</ParamField>

<ParamField path="size" type="'xs' | 'sm' | 'md' | 'lg' | 'xl'" default="'md'">
  The size of the accordion and its children.
</ParamField>

<ParamField path="defaultValue" type="string[]">
  The default open items (uncontrolled mode).
</ParamField>

<ParamField path="value" type="string[]">
  The controlled open items.
</ParamField>

<ParamField path="onValueChange" type="(value: string[]) => void">
  Callback fired when the open items change.
</ParamField>

<ParamField path="hiddenUntilFound" type="boolean">
  Enable browser find-in-page to reveal collapsed content.
</ParamField>

### Accordion.Item

<ParamField path="value" type="string" required>
  A unique value for this accordion item.
</ParamField>

<ParamField path="disabled" type="boolean">
  Whether the item is disabled.
</ParamField>

### Accordion.Trigger

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

### Accordion.Content

<ParamField path="keepMounted" type="boolean">
  Keep content mounted in the DOM when collapsed.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply.
</ParamField>

## Advanced Patterns

### Single Item Mode

```tsx
<Accordion defaultValue={["item-1"]}>
  {/* Only one item can be open at a time */}
</Accordion>
```

### Controlled State

```tsx
const [openItems, setOpenItems] = useState(["item-1"]);

<Accordion value={openItems} onValueChange={setOpenItems}>
  {/* ... */}
</Accordion>
```

### Keep Content Mounted

```tsx
<Accordion.Content keepMounted>
  {/* Content stays in DOM when collapsed */}
</Accordion.Content>
```
